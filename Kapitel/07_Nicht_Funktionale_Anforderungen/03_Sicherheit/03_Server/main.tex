\subsection*{Server}

\begin{samepage}
    \begin{enumerate}[label=\textbf{/NFS\arabic*0/}, align=left, start=9]
        \item Das \Gls{Backend} läuft in einer eigenen \Gls{VLU}.
        \item Das \Gls{Backend} nutzt ausschließlich eine eigene \Gls{Datenbank}. Sie hat den alleinigen \Gls{Zugang} auf diese.
        \item Der \Gls{Datenaustausch} mit der \Gls{Datenbank} erfolgt ausschließlich unmittelbar durch das \Gls{Backend}.
        \item Der \Gls{Server} kommuniziert ausschließlich verschlüsselt über \Gls{HTTPS}.
    \end{enumerate}
\end{samepage}
